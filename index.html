import pygame import math import sys # Инициализация pygame.init() WIDTH, HEIGHT = 1200, 800 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("🚀 Space Flight Simulator BRO") clock = pygame.time.Clock() # Цвета BLACK = (0, 0, 0) WHITE = (255, 255, 255) RED = (255, 50, 50) BLUE = (50, 50, 255) GREEN = (50, 255, 50) YELLOW = (255, 255, 0) GRAY = (100, 100, 100) ORANGE = (255, 150, 0) CYAN = (0, 255, 255) # Физические константы G = 500 # Гравитационная постоянная GROUND_Y = HEIGHT - 50 class CelestialBody: def __init__(self, x, y, radius, mass, color, name): self.x = x self.y = y self.radius = radius self.mass = mass self.color = color self.name = name def draw(self, screen, offset_x, offset_y): pygame.draw.circle(screen, self.color, (int(self.x - offset_x), int(self.y - offset_y)), self.radius) # Атмосфера if self.radius > 30: pygame.draw.circle(screen, (*self.color, 30), (int(self.x - offset_x), int(self.y - offset_y)), self.radius + 20, 3) class Rocket: def __init__(self, x, y): self.x = x self.y = y self.vx = 0 self.vy = 0 self.angle = -math.pi/2 # Направлен вверх self.fuel = 100 self.max_fuel = 100 self.thrust = 0 self.max_thrust = 15 self.mass = 10 self.parts = [] self.is_thrusting = False self.alive = True self.trail = [] # Строим ракету (упрощённо) self.build_default_rocket() def build_default_rocket(self): # Капсула self.parts.append({'type': 'capsule', 'rel_x': 0, 'rel_y': -20, 'w': 14, 'h': 16}) # Топливный бак self.parts.append({'type': 'tank', 'rel_x': 0, 'rel_y': -4, 'w': 16, 'h': 24}) # Двигатель self.parts.append({'type': 'engine', 'rel_x': 0, 'rel_y': 18, 'w': 12, 'h': 10}) def apply_force(self, fx, fy): self.vx += fx / self.mass self.vy += fy / self.mass def update(self, bodies): if not self.alive: return # Гравитация for body in bodies: dx = body.x - self.x dy = body.y - self.y dist = math.sqrt(dx**2 + dy**2) if dist < body.radius: self.alive = False return if dist > 0: force = G * body.mass / (dist**2) self.vx += force * dx / dist self.vy += force * dy / dist # Тяга if self.is_thrusting and self.fuel > 0: thrust_x = math.cos(self.angle) * self.thrust thrust_y = math.sin(self.angle) * self.thrust self.vx += thrust_x / self.mass self.vy += thrust_y / self.mass self.fuel -= 0.2 if self.fuel < 0: self.fuel = 0 # Добавляем след if len(self.trail) > 20: self.trail.pop(0) self.trail.append((self.x, self.y, self.angle)) # Движение self.x += self.vx self.y += self.vy # Проверка столкновения с землей for body in bodies: dx = body.x - self.x dy = body.y - self.y dist = math.sqrt(dx**2 + dy**2) if dist < body.radius: self.alive = False def draw(self, screen, offset_x, offset_y): # Рисуем след for i, (tx, ty, ta) in enumerate(self.trail): alpha = i * 10 if alpha > 255: alpha = 255 x = tx - offset_x y = ty - offset_y pygame.draw.circle(screen, (255, 200, 50, alpha), (int(x), int(y)), 3) # Рисуем части ракеты for part in self.parts: # Поворачиваем относительные координаты rel_x = part['rel_x'] * math.cos(self.angle) - part['rel_y'] * math.sin(self.angle) rel_y = part['rel_x'] * math.sin(self.angle) + part['rel_y'] * math.cos(self.angle) px = self.x + rel_x - offset_x py = self.y + rel_y - offset_y if part['type'] == 'capsule': pygame.draw.ellipse(screen, WHITE, (int(px-part['w']/2), int(py-part['h']/2), part['w'], part['h'])) # Иллюминатор pygame.draw.circle(screen, CYAN, (int(px), int(py-4)), 3) elif part['type'] == 'tank': pygame.draw.rect(screen, ORANGE, (int(px-part['w']/2), int(py-part['h']/2), part['w'], part['h'])) pygame.draw.rect(screen, RED, (int(px-part['w']/2+2), int(py-part['h']/2+2), part['w']-4, part['h']-4)) elif part['type'] == 'engine': pygame.draw.polygon(screen, GRAY, [ (px-part['w']/2, py-part['h']/2), (px+part['w']/2, py-part['h']/2), (px+part['w']/4, py+part['h']/2), (px-part['w']/4, py+part['h']/2) ]) # Рисуем пламя if self.is_thrusting and self.fuel > 0: flame_x = self.x + math.cos(self.angle) * 18 - offset_x flame_y = self.y + math.sin(self.angle) * 18 - offset_y for _ in range(3): flame_len = 15 + math.sin(pygame.time.get_ticks() * 0.02) * 5 fx = flame_x - math.cos(self.angle) * flame_len fy = flame_y - math.sin(self.angle) * flame_len pygame.draw.circle(screen, YELLOW, (int(fx), int(fy)), 4) pygame.draw.circle(screen, RED, (int(fx), int(fy)), 2) class Game: def __init__(self): self.earth = CelestialBody(WIDTH//2, HEIGHT//2 + 200, 80, 1000, BLUE, "Earth") self.moon = CelestialBody(WIDTH//2 + 400, HEIGHT//2 - 100, 25, 100, GRAY, "Moon") self.bodies = [self.earth, self.moon] self.rocket = Rocket(WIDTH//2, HEIGHT//2 + 50) self.camera_x = 0 self.camera_y = 0 self.font = pygame.font.Font(None, 36) self.small_font = pygame.font.Font(None, 24) self.show_trajectory = True self.time_warp = 1 def update_camera(self): self.camera_x = self.rocket.x - WIDTH//2 self.camera_y = self.rocket.y - HEIGHT//2 def handle_input(self): keys = pygame.key.get_pressed() # Управление тягой if keys[pygame.K_w] or keys[pygame.K_UP]: self.rocket.is_thrusting = True self.rocket.thrust = self.rocket.max_thrust elif keys[pygame.K_s] or keys[pygame.K_DOWN]: self.rocket.is_thrusting = True self.rocket.thrust = self.rocket.max_thrust * 0.3 else: self.rocket.is_thrusting = False self.rocket.thrust = 0 # Вращение if keys[pygame.K_a] or keys[pygame.K_LEFT]: self.rocket.angle -= 0.05 if keys[pygame.K_d] or keys[pygame.K_RIGHT]: self.rocket.angle += 0.05 # Time warp if keys[pygame.K_1]: self.time_warp = 1 if keys[pygame.K_2]: self.time_warp = 2 if keys[pygame.K_3]: self.time_warp = 5 if keys[pygame.K_4]: self.time_warp = 10 def update(self): if self.rocket.alive: self.handle_input() for _ in range(self.time_warp): self.rocket.update(self.bodies) self.update_camera() def draw_hud(self): # Индикатор топлива fuel_text = self.font.render(f"⛽ Fuel: {int(self.rocket.fuel)}%", True, WHITE) screen.blit(fuel_text, (10, 10)) # Скорость speed = math.sqrt(self.rocket.vx**2 + self.rocket.vy**2) speed_text = self.font.render(f"🚀 Speed: {speed:.1f} m/s", True, WHITE) screen.blit(speed_text, (10, 50)) # Высота над землей alt = math.sqrt((self.rocket.x - self.earth.x)**2 + (self.rocket.y - self.earth.y)**2) - self.earth.radius alt_text = self.font.render(f"📏 Altitude: {max(0, alt):.1f} km", True, WHITE) screen.blit(alt_text, (10, 90)) # Time warp warp_text = self.font.render(f"⏩ Warp: x{self.time_warp}", True, YELLOW) screen.blit(warp_text, (10, 130)) # Управление controls = [ "W/UP: Full thrust | S/DOWN: Low thrust", "A/LEFT D/RIGHT: Rotate", "1-4: Time warp", "R: Restart" ] y_offset = HEIGHT - 100 for control in controls: ctrl_text = self.small_font.render(control, True, GRAY) screen.blit(ctrl_text, (10, y_offset)) y_offset += 25 def draw(self): screen.fill(BLACK) # Рисуем звезды for i in range(100): x = (i * 137 + 50) % WIDTH y = (i * 97 + 30) % HEIGHT if i % 3 == 0: pygame.draw.circle(screen, WHITE, (x, y), 2) else: pygame.draw.circle(screen, (100, 100, 100), (x, y), 1) # Рисуем небесные тела for body in self.bodies: body.draw(screen, self.camera_x, self.camera_y) # Рисуем траекторию if self.show_trajectory and self.rocket.alive: sim_x, sim_y = self.rocket.x, self.rocket.y sim_vx, sim_vy = self.rocket.vx, self.rocket.vy for i in range(100): # Гравитация for body in self.bodies: dx = body.x - sim_x dy = body.y - sim_y dist = math.sqrt(dx**2 + dy**2) if dist > body.radius: force = G * body.mass / (dist**2) sim_vx += force * dx / dist * 0.5 sim_vy += force * dy / dist * 0.5 sim_x += sim_vx * 0.5 sim_y += sim_vy * 0.5 if i % 3 == 0: px = int(sim_x - self.camera_x) py = int(sim_y - self.camera_y) if 0 <= px < WIDTH and 0 <= py < HEIGHT: alpha = 200 - i * 2 if alpha > 0: pygame.draw.circle(screen, (100, 200, 255, alpha), (px, py), 1) # Рисуем ракету self.rocket.draw(screen, self.camera_x, self.camera_y) # HUD self.draw_hud() # Смерть if not self.rocket.alive: game_over = self.font.render("💥 ROCKET DESTROYED! Press R to restart", True, RED) text_rect = game_over.get_rect(center=(WIDTH//2, HEIGHT//2)) screen.blit(game_over, text_rect) pygame.display.flip() def run(self): running = True while running: for event in pygame.event.get(): if event.type == pygame.QUIT: running = False elif event.type == pygame.KEYDOWN: if event.key == pygame.K_r: self.rocket = Rocket(WIDTH//2, HEIGHT//2 + 50) elif event.key == pygame.K_t: self.show_trajectory = not self.show_trajectory elif event.key == pygame.K_ESCAPE: running = False self.update() self.draw() clock.tick(60) pygame.quit() sys.exit() # Запуск if __name__ == "__main__": game = Game() game.run()
