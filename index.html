import numpy as np import matplotlib.pyplot as plt from matplotlib.animation import FuncAnimation import matplotlib.patches as patches from matplotlib.path import Path # Создаем фигуру fig, ax = plt.subplots(figsize=(10, 8)) ax.set_xlim(0, 10) ax.set_ylim(0, 10) ax.set_aspect('equal') ax.set_facecolor('#f0f8f0') # Легкий зеленый фон # Кактус в центре def draw_cactus(x, y): # Ствол ax.add_patch(patches.Rectangle((x-0.3, y), 0.6, 3, facecolor='#2e8b57', edgecolor='darkgreen')) # Руки кактуса ax.add_patch(patches.Rectangle((x-1.2, y+1.2), 0.8, 1.5, facecolor='#2e8b57', edgecolor='darkgreen')) ax.add_patch(patches.Rectangle((x+0.4, y+1.5), 0.8, 1.2, facecolor='#2e8b57', edgecolor='darkgreen')) # Колючки for i in range(5): ax.plot([x-0.4, x-0.6], [y+0.5+i*0.5, y+0.6+i*0.5], 'k-', lw=1) ax.plot([x+0.4, x+0.6], [y+0.5+i*0.5, y+0.6+i*0.5], 'k-', lw=1) draw_cactus(5, 2) # Простые слоны (используем круги + треугольники) def draw_elephant(x, y, color='gray', angle=0): # Тело body = patches.Ellipse((x, y), 1.8, 1.2, angle=angle, facecolor=color, edgecolor='black', lw=2) ax.add_patch(body) # Голова head = patches.Circle((x+0.9, y+0.3), 0.7, facecolor=color, edgecolor='black', lw=2) ax.add_patch(head) # Хобот trunk_path = Path([(x+1.5, y+0.2), (x+2.0, y-0.3), (x+2.1, y-0.6)], [Path.MOVETO, Path.CURVE3, Path.CURVE3]) trunk = patches.PathPatch(trunk_path, facecolor='none', edgecolor='black', lw=4) ax.add_patch(trunk) # Ухо ear = patches.Ellipse((x+0.6, y+0.7), 0.6, 0.8, angle=30, facecolor=color, edgecolor='black') ax.add_patch(ear) # Ноги ax.add_patch(patches.Rectangle((x-0.6, y-0.7), 0.4, 0.8, facecolor=color, edgecolor='black')) ax.add_patch(patches.Rectangle((x+0.2, y-0.7), 0.4, 0.8, facecolor=color, edgecolor='black')) # Глаз ax.plot(x+1.2, y+0.4, 'ko', markersize=6) ax.plot(x+1.25, y+0.42, 'wo', markersize=3) # Два слона ele1_x, ele1_y = [], [] ele2_x, ele2_y = [], [] def animate(frame): ax.clear() ax.set_xlim(0, 10) ax.set_ylim(0, 10) ax.set_aspect('equal') ax.set_facecolor('#f0f8f0') # Перерисовываем кактус draw_cactus(5, 2) # Слон 1 бегает по кругу (радиус 3.5) angle1 = frame * 0.08 x1 = 5 + 3.5 * np.cos(angle1) y1 = 4.5 + 3.5 * np.sin(angle1) * 0.7 # немного овальный круг draw_elephant(x1, y1, color='#808080', angle=np.degrees(angle1) + 90) # Слон 2 бегает по другому кругу (радиус 4.2, в противоположную сторону) angle2 = -frame * 0.06 + np.pi x2 = 5 + 4.2 * np.cos(angle2) y2 = 4.5 + 4.2 * np.sin(angle2) * 0.7 draw_elephant(x2, y2, color='#696969', angle=np.degrees(angle2) + 90) # Текст ax.text(5, 9.2, 'Два слона бегают вокруг кактуса!', ha='center', fontsize=14, fontweight='bold') # Следы/пыль (простые точки) if frame > 5: ax.scatter([x1-1.5, x2-1.5], [y1-0.8, y2-0.8], c='brown', s=40, alpha=0.4) ani = FuncAnimation(fig, animate, frames=180, interval=50, repeat=True) # Сохраняем видео ani.save('elephants_around_cactus.mp4', writer='ffmpeg', fps=20, dpi=120) print("Видео успешно создано: elephants_around_cactus.mp4") plt.close()
